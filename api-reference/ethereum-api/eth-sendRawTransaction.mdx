# eth_sendRawTransaction

Creates a new message call transaction or a contract creation for signed transactions

<ParamField body="body" type="object">
  <ParamField body="jsonrpc" type="string" default="2.0" required>
    JSON-RPC version
  </ParamField>
  <ParamField body="method" type="string" default="eth_sendRawTransaction" required>
    Method name
  </ParamField>
  <ParamField body="id" type="integer" default="1" required>
    Request identifier
  </ParamField>
  <ParamField body="params" type="array" required>
    <ParamField body="0" type="string" required>
      Signed transaction data as hex string
    </ParamField>
  </ParamField>
</ParamField>

<ResponseField name="result" type="string">
  Transaction hash (32-byte hex string)
</ResponseField>

## Example Request

<CodeGroup>

```bash cURL
curl --request POST \
  --url https://rpc.trex.xyz/ \
  --header 'Content-Type: application/json' \
  --data '{
  "jsonrpc": "2.0",
  "id": 1,
  "method": "eth_sendRawTransaction",
  "params": [
    "0xf86c0a8502540be400825208943535353535353535353535353535353535353535880de0b6b3a76400008025a028ef61340bd939bc2195fe537567866003e1a15d3c71ff63e1590620aa636276a067cbe9d8997f761aecb703304b3800ccf555c9f3dc64214b297fb1966a3b6d83"
  ]
}'
```

```javascript JavaScript
const { ethers } = require('ethers');

const provider = new ethers.providers.JsonRpcProvider('https://rpc.trex.xyz/');
const wallet = new ethers.Wallet(process.env.PRIVATE_KEY, provider);

const tx = await wallet.sendTransaction({
  to: '0x3535353535353535353535353535353535353535',
  value: ethers.utils.parseEther('1.0')
});

const txHash = tx.hash;
console.log('Transaction hash:', txHash);
```

</CodeGroup>
